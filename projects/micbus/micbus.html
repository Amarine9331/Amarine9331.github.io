1 Scope

This notebook is a report-grade V2 local ovarian HGSOC run for testing whether MiCBuS can recover marker genes for a tumor microenvironment population that is present in bulk tumors but intentionally omitted from the single-cell reference.

The default missing population is stromal_caf. The expected positive controls are COL1A1, COL1A2, DCN, LUM, FAP, ACTA2, PDGFRA, and PDGFRB. An optional second holdout is myeloid, with expected markers LYZ, C1QA, C1QB, C1QC, CD68, APOE, and SPP1.

This workflow uses only public processed expression data:

  • GSE211956 processed GEO files for five scRNA-seq samples and two Visium sections.
  • Open TCGA-OV summarized gene expression.
  • No raw reads, dbGaP, EGA, low-pass WGS, CNA calls, or restricted data.

`%||%` <- function(x, y) {
  if (is.null(x) || length(x) == 0) y else x
}

safe_genes <- function(genes, object_or_matrix) {
  available <- if (inherits(object_or_matrix, "Seurat")) {
    rownames(object_or_matrix)
  } else {
    rownames(object_or_matrix)
  }
  intersect(genes, available)
}

join_rna_layers <- function(seurat_object) {
  join_fn <- NULL
  if ("JoinLayers" %in% getNamespaceExports("Seurat")) {
    join_fn <- getExportedValue("Seurat", "JoinLayers")
  } else if (requireNamespace("SeuratObject", quietly = TRUE) &&
             "JoinLayers" %in% getNamespaceExports("SeuratObject")) {
    join_fn <- getExportedValue("SeuratObject", "JoinLayers")
  }

  if (!is.null(join_fn)) {
    seurat_object <- join_fn(seurat_object, assay = "RNA")
  }
  seurat_object
}

}

read_count_txt <- function(path) {
  dt <- data.table::fread(path)
  gene_col <- names(dt)[1]
  genes <- make.unique(as.character(dt[[gene_col]]))
  mat <- as.matrix(dt[, -1, with = FALSE])
  rownames(mat) <- genes
  Matrix::Matrix(mat, sparse = TRUE)
}

sample_id_from_path <- function(path) {
  base <- basename(path)
  sub("^GSM[0-9]+_counts_([^.]*)\\.txt\\.gz$", "\\1", base)
}

downsample_by_class <- function(meta, class_col, max_per_class, seed = 1) {
  set.seed(seed)
  split(seq_len(nrow(meta)), meta[[class_col]]) |>
    lapply(function(idx) sample(idx, min(length(idx), max_per_class))) |>
    unlist(use.names = FALSE)
}

average_expression_by_class <- function(seurat_object, class_col = "broad_class") {
  avg <- Seurat::AverageExpression(
    seurat_object,
    group.by = class_col,
    assays = "RNA",
    slot = "data"
  )$RNA
  as.matrix(avg)
}

score_gene_set_matrix <- function(expr, genes) {
  genes <- intersect(genes, rownames(expr))
  if (length(genes) == 0) {
    stop("No marker genes were found in the expression matrix.")
  }
  colMeans(expr[genes, , drop = FALSE])
}

normalize_bulk_matrix <- function(counts_or_expression) {
  mat <- as.matrix(counts_or_expression)
  mat <- mat[rowSums(is.finite(mat)) == ncol(mat), , drop = FALSE]
  mat <- mat[!duplicated(rownames(mat)), , drop = FALSE]
  if (max(mat, na.rm = TRUE) > 100) {
    lib <- colSums(mat)
    mat <- t(t(mat) / lib * 1e6)
    mat <- log2(mat + 1)
  }
  mat
}

query_tcga_ov_counts <- function(n_samples, seed = 1) {
  workflows <- c("STAR - Counts", "HTSeq - Counts")
  last_error <- NULL

  for (workflow in workflows) {
    query <- tryCatch(
      TCGAbiolinks::GDCquery(
        project = "TCGA-OV",
        data.category = "Transcriptome Profiling",
        data.type = "Gene Expression Quantification",
        workflow.type = workflow,
        sample.type = "Primary Tumor"
      ),
      error = function(e) e
    )

    if (inherits(query, "error")) {
      last_error <- query
      next
    }

    cases_result <- TCGAbiolinks::getResults(query, cols = "cases")
    cases <- if (is.data.frame(cases_result)) cases_result$cases else cases_result
    cases <- unique(cases)
    cases <- cases[!is.na(cases)]
    if (length(cases) == 0) {
      next
    }

    set.seed(seed)
    selected_cases <- sample(cases, min(n_samples, length(cases)))

    subset_query <- TCGAbiolinks::GDCquery(
      project = "TCGA-OV",
      data.category = "Transcriptome Profiling",
      data.type = "Gene Expression Quantification",
      workflow.type = workflow,
      sample.type = "Primary Tumor",
      barcode = selected_cases
    )

    return(subset_query)
  }

  stop("Unable to create a TCGA-OV query. Last error: ", last_error$message %||% "none")
}

simulate_dirichlet_weights <- function(n, k, alpha = 1) {
  raw <- matrix(rgamma(n * k, shape = alpha, rate = 1), nrow = n, ncol = k)
  raw / rowSums(raw)
}

run_micbus_fallback <- function(bulk_expr, reference_profiles, n_reps, seed = 1) {
  set.seed(seed)
  genes <- Reduce(intersect, list(rownames(bulk_expr), rownames(reference_profiles)))
  bulk <- bulk_expr[genes, , drop = FALSE]
  ref <- reference_profiles[genes, , drop = FALSE]

  weights <- simulate_dirichlet_weights(n_reps, ncol(ref), alpha = 1)
  pseudo <- ref %*% t(weights)
  colnames(pseudo) <- paste0("pseudo_", seq_len(n_reps))

  combined <- cbind(bulk, pseudo)
  group <- factor(c(rep("bulk", ncol(bulk)), rep("pseudo", ncol(pseudo))), levels = c("pseudo", "bulk"))

  res <- lapply(seq_len(nrow(combined)), function(i) {
    fit <- tryCatch(
      stats::t.test(combined[i, group == "bulk"], combined[i, group == "pseudo"]),
      error = function(e) NULL
    )
    if (is.null(fit)) {
      return(data.frame(gene = rownames(combined)[i], logFC = NA_real_, p_value = NA_real_))
    }
    data.frame(
      gene = rownames(combined)[i],
      logFC = mean(combined[i, group == "bulk"]) - mean(combined[i, group == "pseudo"]),
      p_value = fit$p.value
    )
  }) |>
    dplyr::bind_rows() |>
    dplyr::mutate(p_adj = p.adjust(p_value, method = "BH")) |>
    dplyr::arrange(dplyr::desc(logFC), p_adj)

  res
}

run_micbus <- function(bulk_expr, incomplete_reference, n_reps, seed = 1) {
  # Prefer a user-installed MiCBuS package when it exposes a recognizable entry point.
  if (requireNamespace("MiCBuS", quietly = TRUE)) {
    ns <- asNamespace("MiCBuS")
    exported <- getNamespaceExports("MiCBuS")
    candidate <- intersect(c("MiCBuS", "micbus", "run_micbus", "runMiCBuS"), exported)

    if (length(candidate) > 0) {
      fn <- get(candidate[1], envir = ns)
      message("Running MiCBuS package function: MiCBuS::", candidate[1])
      out <- tryCatch(
        fn(
          bulk = bulk_expr,
          sc_ref = incomplete_reference,
          n_pseudobulk = n_reps,
          seed = seed
        ),
        error = function(e) e
      )
      if (!inherits(out, "error")) {
        return(as.data.frame(out))
      }
      message("Installed MiCBuS call failed; using tutorial fallback. Error: ", out$message)
    }
  }

  message("Using tutorial fallback: Dirichlet pseudo-bulk plus bulk-vs-pseudo differential test.")
  run_micbus_fallback(bulk_expr, incomplete_reference, n_reps = n_reps, seed = seed)
}

4 Build the V2 local scRNA reference

The GEO scRNA files are processed count matrices without harmonized broad cell class labels. This report assigns broad labels by clustering the merged object and selecting the strongest canonical marker module for each cluster.

scrna_cache <- file.path(
  "data/cache",
  paste0("gse211956_scrna_", params$mode, "_", max_cells_per_class, ".rds")
)

if (file.exists(scrna_cache)) {
  scrna <- readRDS(scrna_cache)
} else {
  if (length(scrna_count_files) == 0) {
    stop("No GSE211956 scRNA count files were found after GEO download.")
  }

  objects <- lapply(scrna_count_files, function(path) {
    sample_id <- sample_id_from_path(path)
    counts <- read_count_txt(path)
    obj <- Seurat::CreateSeuratObject(counts, project = sample_id, min.cells = 3, min.features = 200)
    obj$sample_id <- sample_id
    obj[["percent.mt"]] <- Seurat::PercentageFeatureSet(obj, pattern = "^MT-")
    subset(obj, subset = nFeature_RNA >= 500 & nCount_RNA >= 1000 & percent.mt < 20)
  })

  scrna <- Reduce(function(x, y) merge(x, y), objects)
  scrna <- Seurat::NormalizeData(scrna)
  scrna <- Seurat::FindVariableFeatures(scrna, nfeatures = 2500)
  scrna <- Seurat::ScaleData(scrna, features = Seurat::VariableFeatures(scrna))
  scrna <- Seurat::RunPCA(scrna, features = Seurat::VariableFeatures(scrna), npcs = 30)
  scrna <- Seurat::FindNeighbors(scrna, dims = 1:20)
  scrna <- Seurat::FindClusters(scrna, resolution = 0.5)
  scrna <- Seurat::RunUMAP(scrna, dims = 1:20)

  for (nm in names(marker_sets)) {
    genes <- safe_genes(marker_sets[[nm]], scrna)
    if (length(genes) > 0) {
      scrna <- Seurat::AddModuleScore(scrna, features = list(genes), name = paste0(nm, "_score"))
    }
  }

  score_cols <- paste0(names(marker_sets), "_score1")
  cluster_scores <- scrna@meta.data |>
    tibble::rownames_to_column("cell") |>
    dplyr::group_by(seurat_clusters) |>
    dplyr::summarise(dplyr::across(dplyr::all_of(score_cols), mean), .groups = "drop")

  cluster_label <- apply(cluster_scores[, score_cols, drop = FALSE], 1, function(x) {
    names(marker_sets)[which.max(x)]
  })
  names(cluster_label) <- cluster_scores$seurat_clusters

  scrna$broad_class <- unname(cluster_label[as.character(scrna$seurat_clusters)])

  keep_idx <- downsample_by_class(
    scrna@meta.data,
    class_col = "broad_class",
    max_per_class = max_cells_per_class,
    seed = params$seed
  )
  scrna <- subset(scrna, cells = colnames(scrna)[keep_idx])
  scrna <- join_rna_layers(scrna)

  saveRDS(scrna, scrna_cache)
}

# Scope

This notebook is a report-grade V2 local ovarian HGSOC run for testing whether
MiCBuS can recover marker genes for a tumor microenvironment population that is
present in bulk tumors but intentionally omitted from the single-cell reference.

The default missing population is `stromal_caf`. The expected positive controls
are `COL1A1`, `COL1A2`, `DCN`, `LUM`, `FAP`, `ACTA2`, `PDGFRA`, and `PDGFRB`.
An optional second holdout is `myeloid`, with expected markers `LYZ`, `C1QA`,
`C1QB`, `C1QC`, `CD68`, `APOE`, and `SPP1`.

This workflow uses only public processed expression data:

- `GSE211956` processed GEO files for five scRNA-seq samples and two
  Visium sections.
- Open TCGA-OV summarized gene expression.
- No raw reads, dbGaP, EGA, low-pass WGS, CNA calls, or restricted data.

```{r helpers}
normalize_bulk_matrix <- function(counts_or_expression) {
  mat <- as.matrix(counts_or_expression)
  mat <- mat[rowSums(is.finite(mat)) == ncol(mat), , drop = FALSE]
  mat <- mat[!duplicated(rownames(mat)), , drop = FALSE]
  if (max(mat, na.rm = TRUE) > 100) {
    lib <- colSums(mat)
    mat <- t(t(mat) / lib * 1e6)
    mat <- log2(mat + 1)
  }
  mat
}

query_tcga_ov_counts <- function(n_samples, seed = 1) {
  workflows <- c("STAR - Counts", "HTSeq - Counts")
  last_error <- NULL

  for (workflow in workflows) {

        project = "TCGA-OV",
        data.category = "Transcriptome Profiling",
        data.type = "Gene Expression Quantification",
        workflow.type = workflow,
        sample.type = "Primary Tumor"
      ),
      error = function(e) e
    )

    if (inherits(query, "error")) {
      last_error <- query
      next
    }

    cases_result <- TCGAbiolinks::getResults(query, cols = "cases")
    cases <- if (is.data.frame(cases_result)) cases_result$cases else cases_result
    cases <- unique(cases)
    cases <- cases[!is.na(cases)]
    if (length(cases) == 0) {
      next
    }

    set.seed(seed)
    selected_cases <- sample(cases, min(n_samples, length(cases)))

    subset_query <- TCGAbiolinks::GDCquery(

      data.type = "Gene Expression Quantification",

}

simulate_dirichlet_weights <- function(n, k, alpha = 1) {
  raw <- matrix(rgamma(n * k, shape = alpha, rate = 1), nrow = n, ncol = k)
  raw / rowSums(raw)
}

run_micbus_fallback <- function(bulk_expr, reference_profiles, n_reps, seed = 1) {
  set.seed(seed)
  genes <- Reduce(intersect, list(rownames(bulk_expr), rownames(reference_profiles)))
  bulk <- bulk_expr[genes, , drop = FALSE]
  ref <- reference_profiles[genes, , drop = FALSE]

  weights <- simulate_dirichlet_weights(n_reps, ncol(ref), alpha = 1)
  pseudo <- ref %*% t(weights)
  colnames(pseudo) <- paste0("pseudo_", seq_len(n_reps))

  combined <- cbind(bulk, pseudo)
  group <- factor(c(rep("bulk", ncol(bulk)), rep("pseudo", ncol(pseudo))), levels = c("pseudo", "bulk"))

  res <- lapply(seq_len(nrow(combined)), function(i) {
    fit <- tryCatch(
      stats::t.test(combined[i, group == "bulk"], combined[i, group == "pseudo"]),
      error = function(e) NULL
    )
    if (is.null(fit)) {
      return(data.frame(gene = rownames(combined)[i], logFC = NA_real_, p_value = NA_real_))
    }
    data.frame(
      gene = rownames(combined)[i],
      logFC = mean(combined[i, group == "bulk"]) - mean(combined[i, group == "pseudo"]),
      p_value = fit$p.value
    )
  }) |>
    dplyr::bind_rows() |>
    dplyr::mutate(p_adj = p.adjust(p_value, method = "BH")) |>
    dplyr::arrange(dplyr::desc(logFC), p_adj)

  res
}

run_micbus <- function(bulk_expr, incomplete_reference, n_reps, seed = 1) {
  # Prefer a user-installed MiCBuS package when it exposes a recognizable entry point.
  if (requireNamespace("MiCBuS", quietly = TRUE)) {
    ns <- asNamespace("MiCBuS")
    exported <- getNamespaceExports("MiCBuS")
    candidate <- intersect(c("MiCBuS", "micbus", "run_micbus", "runMiCBuS"), exported)

    if (length(candidate) > 0) {
      fn <- get(candidate[1], envir = ns)
      message("Running MiCBuS package function: MiCBuS::", candidate[1])
      out <- tryCatch(
        fn(
          bulk = bulk_expr,
          sc_ref = incomplete_reference,
          n_pseudobulk = n_reps,
          seed = seed
        ),
        error = function(e) e
      )
      if (!inherits(out, "error")) {
        return(as.data.frame(out))
      }
      message("Installed MiCBuS call failed; using tutorial fallback. Error: ", out$message)
    }
  }

  message("Using tutorial fallback: Dirichlet pseudo-bulk plus bulk-vs-pseudo differential test.")
  run_micbus_fallback(bulk_expr, incomplete_reference, n_reps = n_reps, seed = seed)
}
```

# Build the V2 local scRNA reference

The GEO scRNA files are processed count matrices without harmonized broad cell
class labels. This report assigns broad labels by clustering the merged object
and selecting the strongest canonical marker module for each cluster.

```{r scrna-reference}
scrna_cache <- file.path(
  "data/cache",
  paste0("gse211956_scrna_", params$mode, "_", max_cells_per_class, ".rds")
)

if (file.exists(scrna_cache)) {
  scrna <- readRDS(scrna_cache)
} else {
  if (length(scrna_count_files) == 0) {
    stop("No GSE211956 scRNA count files were found after GEO download.")
  }

  objects <- lapply(scrna_count_files, function(path) {
    sample_id <- sample_id_from_path(path)
    counts <- read_count_txt(path)
    obj <- Seurat::CreateSeuratObject(counts, project = sample_id, min.cells = 3, min.features = 200)
    obj$sample_id <- sample_id
    obj[["percent.mt"]] <- Seurat::PercentageFeatureSet(obj, pattern = "^MT-")
    subset(obj, subset = nFeature_RNA >= 500 & nCount_RNA >= 1000 & percent.mt < 20)
  })

  scrna <- Reduce(function(x, y) merge(x, y), objects)
  scrna <- Seurat::NormalizeData(scrna)
  scrna <- Seurat::FindVariableFeatures(scrna, nfeatures = 2500)
  scrna <- Seurat::ScaleData(scrna, features = Seurat::VariableFeatures(scrna))
  scrna <- Seurat::RunPCA(scrna, features = Seurat::VariableFeatures(scrna), npcs = 30)
  scrna <- Seurat::FindNeighbors(scrna, dims = 1:20)
  scrna <- Seurat::FindClusters(scrna, resolution = 0.5)
  scrna <- Seurat::RunUMAP(scrna, dims = 1:20)

  for (nm in names(marker_sets)) {
    genes <- safe_genes(marker_sets[[nm]], scrna)
    if (length(genes) > 0) {
      scrna <- Seurat::AddModuleScore(scrna, features = list(genes), name = paste0(nm, "_score"))
    }
  }

  score_cols <- paste0(names(marker_sets), "_score1")
  cluster_scores <- scrna@meta.data |>
    tibble::rownames_to_column("cell") |>
    dplyr::group_by(seurat_clusters) |>
    dplyr::summarise(dplyr::across(dplyr::all_of(score_cols), mean), .groups = "drop")

  cluster_label <- apply(cluster_scores[, score_cols, drop = FALSE], 1, function(x) {
    names(marker_sets)[which.max(x)]
  })
  names(cluster_label) <- cluster_scores$seurat_clusters

  scrna$broad_class <- unname(cluster_label[as.character(scrna$seurat_clusters)])

  keep_idx <- downsample_by_class(
    scrna@meta.data,
    class_col = "broad_class",
    max_per_class = max_cells_per_class,
    seed = params$seed
  )
  scrna <- subset(scrna, cells = colnames(scrna)[keep_idx])
  scrna <- join_rna_layers(scrna)

  saveRDS(scrna, scrna_cache)
}
```

```{r tcga-download}

  bulk_expr <- readRDS(tcga_cache)
} else {

  }

  TCGAbiolinks::GDCdownload(query, directory = "data/raw/TCGA_OV")
  se <- TCGAbiolinks::GDCprepare(query, directory = "data/raw/TCGA_OV")

  counts <- SummarizedExperiment::assay(se)
  rd <- as.data.frame(SummarizedExperiment::rowData(se))
  gene_symbols <- rd$external_gene_name %||% rd$gene_name %||% rd$gene_id
  keep <- !is.na(gene_symbols) & gene_symbols != ""
  counts <- counts[keep, , drop = FALSE]
  gene_symbols <- gene_symbols[keep]

  counts_by_symbol <- rowsum(as.matrix(counts), group = gene_symbols)
```

# Spatial validation with two Visium sections

The Visium step scores spots for the top pseudo-marker panel and plots the score
in tissue coordinates. This is spatial expression validation only; it does not
make CNA or subclone claims.

```{r load-visium}
load_visium_sample <- function(prefix, geo_dir) {
  sample_files <- list.files(geo_dir, pattern = paste0("_", prefix, "_"), full.names = TRUE)
  matrix_file <- sample_files[grepl("_matrix\\.mtx\\.gz$", sample_files)][1]
  feature_file <- sample_files[grepl("_features\\.tsv\\.gz$", sample_files)][1]
  barcode_file <- sample_files[grepl("_barcodes\\.tsv\\.gz$", sample_files)][1]
  spatial_zip <- sample_files[grepl("_spatial\\.zip$", sample_files)][1]

  if (any(is.na(c(matrix_file, feature_file, barcode_file, spatial_zip)))) {
    stop("Missing Visium processed files for ", prefix)
  }

  spatial_dir <- file.path(geo_dir, paste0(prefix, "_spatial"))
  if (!dir.exists(spatial_dir)) {
    dir.create(spatial_dir, recursive = TRUE)
    utils::unzip(spatial_zip, exdir = spatial_dir)
  }

  mat <- Matrix::readMM(gzfile(matrix_file))
  features <- data.table::fread(feature_file, header = FALSE)
  barcodes <- data.table::fread(barcode_file, header = FALSE)
  rownames(mat) <- make.unique(features[[2]])
  colnames(mat) <- barcodes[[1]]

  obj <- Seurat::CreateSeuratObject(mat, project = prefix, min.features = 100)
  obj <- Seurat::NormalizeData(obj)

  tissue_positions <- list.files(spatial_dir, pattern = "tissue_positions.*\\.csv$", recursive = TRUE, full.names = TRUE)[1]
  if (!is.na(tissue_positions)) {
    pos <- data.table::fread(tissue_positions, header = FALSE)
    if (ncol(pos) >= 6) {
      colnames(pos)[1:6] <- c("barcode", "in_tissue", "array_row", "array_col", "pxl_row", "pxl_col")
      pos <- as.data.frame(pos)
      rownames(pos) <- pos$barcode
      obj <- AddMetaData(obj, pos[colnames(obj), c("in_tissue", "array_row", "array_col", "pxl_row", "pxl_col")])
    }
  }

  obj$visium_sample <- prefix
  obj
}
```
